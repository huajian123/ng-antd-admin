trackBy

使用 trackBy 后，打乱顺序时 Angular 只移动 DOM 节点而不重建，大幅减少渲染开销。

@for (item of items(); track trackById($index, item)) { {{ item.name }} } 
// ✅ 使用 trackBy：打乱时只移动 DOM
          @for (item of items(); track trackById($index, item)) { ... }
          trackById = (_: number, item: Item) => item.id;

// ❌ 不用 trackBy：打乱时销毁重建所有 DOM
          @for (item of items(); track $index) { ... }
computed 缓存

computed() 有缓存：依赖未变化时，多次读取只计算一次。

{{ bigList().length }} {{ expensiveSum() }} {{ actualComputeCount() }} 
readonly expensiveSum = computed(() => {{ '{' }}
          return bigList().reduce((a, b) => a + b, 0);
          {{ '}' }});

// 读取 10 次 expensiveSum()
// → 实际只计算 1 次（缓存命中）
// 依赖 bigList 变化 → 才重新计算
OnPush

OnPush 组件中，Signal 变化自动触发视图更新；普通属性修改不触发。

{{ signalValue() }} {{ plainValue }} 
// ✅ Signal 变化 → OnPush 自动更新
readonly signalValue = signal('初始值');
updateSignal(): void {{ '{' }} this.signalValue.set('新值'); {{ '}' }}

          // ❌ 普通属性修改 → OnPush 不更新视图
plainValue = '初始值';
updatePlain(): void {{ '{' }} this.plainValue = '新值'; {{ '}' }}
untracked()

untracked() 读取 signal 但不建立依赖，b 变化时 withUntracked 不重算。

{{ a() }} {{ b() }} {{ withTracking() }} {{ withUntracked() }} {{ trackingCallCount() }} {{ untrackedCallCount() }} 
// 依赖 a 和 b，b 变化时重算
readonly withTracking = computed(() => a() + b());

// 只依赖 a，b 变化时不重算
readonly withUntracked = computed(() =>
  a() + untracked(() => b())
);
